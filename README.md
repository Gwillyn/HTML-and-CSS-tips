  max-width: 300px;
  height: 50px;
}

  display: flex;
  justify-content: space-evenly;
}

  color: #dfdfe2;
  margin: 0 0.2rem;
  padding: 0.2rem;
  display: block;
}

nav > ul > li:hover {
  background-color: #dfdfe2;
  color: #1b1b32;
  cursor: pointer;
}
